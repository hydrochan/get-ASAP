class BaseParser(ABC):

```python
import logging
import re
from bs4 import BeautifulSoup
from models import PaperMetadata
from parsers.base import BaseParser

logger = logging.getLogger(__name__)
DOI_RE = re.compile(r'10\.\d{4,9}/[^\s"<>#?&]+')

class ACSParser(BaseParser):
    publisher_name = "ACS Publications"

    def can_parse(self, sender: str, subject: str) -> bool:
        return sender == "alerts@acs.org"  # publishers.json 값 (Plan 01에서 검증됨)

    def parse(self, message_body: str) -> list[PaperMetadata]:
        try:
            soup = BeautifulSoup(message_body, "lxml")
            papers = []
            seen_dois = set()  # 중복 DOI 방지 (Pitfall 2 대응)

            # CSS selector: 실제 fixture HTML 구조를 분석하여 확정
            # 폴백 selector 순서: doi.org 링크 -> /doi/ 경로 링크
            for selector in ['a[href*="doi.org"]', 'a[href*="/doi/10."]']:
                links = soup.select(selector)
                if links:
                    break
            else:
                links = []

            for link in links:
                doi = self._extract_doi(link.get("href", ""))
                if not doi or doi in seen_dois:
                    continue
                seen_dois.add(doi)
                title = self._clean_title(link)
                if not title:
                    continue
                papers.append(PaperMetadata(
                    title=title, doi=doi, journal="", date=""
                ))
            return papers
        except Exception as e:
            logger.warning("ACS 파싱 실패: %s", e)
            return []

    @staticmethod
    def _extract_doi(href: str) -> str:
        m = DOI_RE.search(href)
        return m.group().rstrip(".,;)") if m else ""

    @staticmethod
    def _clean_title(tag) -> str:
        import re as _re
        text = tag.get_text(separator=" ", strip=True)
        return _re.sub(r'\s+', ' ', text).strip()
```
